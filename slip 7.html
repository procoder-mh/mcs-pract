## Q2) Event Driven Application // file: main.js // Import EventEmitter module const EventEmitter = require('events'); // Create EventEmitter object const eventEmitter = new EventEmitter(); // Callback Function const eventHandler = () => { console.log("Event detected!"); console.log("Callback function executed successfully."); }; // Listen for event named 'message' eventEmitter.on('message', eventHandler); // Main Loop Simulation console.log("Application is running..."); console.log("Waiting for events..."); // Trigger event after 3 seconds setTimeout(() => { console.log("Event Occurred!"); // Emit event eventEmitter.emit('message'); }, 3000);
