# Slip 9 ## Q1) Concatenate Two Strings ```javascript const http = require('http'); const url = require('url'); http.createServer(function(req, res){ let q = url.parse(req.url, true).query; let str1 = q.str1 || ''; let str2 = q.str2 || ''; res.writeHead(200, {'Content-Type':'text/html'}); res.write('Concatenated String: ' + str1 + str2); res.end(); }).listen(3000); ``` ## Q2) Open Requested File and Return Content ```javascript const http = require('http'); const fs = require('fs'); const url = require('url'); http.createServer(function(req, res){ let q = url.parse(req.url, true); let filename = '.' + q.pathname; fs.readFile(filename, function(err, data){ if(err){ res.writeHead(404, {'Content-Type':'text/html'}); return res.end('404 File Not Found'); } res.writeHead(200, {'Content-Type':'text/html'}); res.write(data); return res.end(); }); }).listen(3000); ``` ---
