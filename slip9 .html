# Slip 9 ## Q1) Concatenate Two Strings // file: concatenate.js // Import HTTP module const http = require('http'); const querystring = require('querystring'); // Create Server http.createServer((req, res) => { // Display HTML Form if (req.method === "GET") { res.writeHead(200, { 'Content-Type': 'text/html' }); res.write(`

Concatenate Two Strings

Enter First String:

Enter Second String:

`); res.end(); } // Handle Form Submission else if (req.method === "POST") { let body = ""; req.on("data", chunk => { body += chunk.toString(); }); req.on("end", () => { // Parse Form Data let data = querystring.parse(body); let str1 = data.str1; let str2 = data.str2; // Concatenate Strings let result = str1 + str2; // Display Result res.writeHead(200, { 'Content-Type': 'text/html' }); res.write(`

Concatenated String

Result: ${result}

`); res.end(); }); } }).listen(3000); console.log("Server running at http://localhost:3000"); ## Q2) Open Requested File and Return Content / file: server.js // Import Modules const http = require('http'); const fs = require('fs'); const url = require('url'); // Create Server http.createServer((req, res) => { // Parse URL let q = url.parse(req.url, true); // Requested File Name let filename = "." + q.pathname; // Read File fs.readFile(filename, function(err, data) { // Handle Error if (err) { res.writeHead(404, { 'Content-Type': 'text/html' }); return res.end("404 Error: File Not Found"); } // Display File Content res.writeHead(200, { 'Content-Type': 'text/html' }); res.write(data); return res.end(); }); }).listen(3000); console.log("Server running at http://localhost:3000"); Example HTML File (index.html) Home Page
