# Slip 2 ## Q1) Reverse String in Node.js // file: reverse.js // Original String let text = "Full Stack!"; // Reverse String let reversedText = text.split("").reverse().join(""); // Display Output console.log("Original String: " + text); console.log("Reversed String: " + reversedText); ## Q2) Append One File into Another // file: appendfile.js // Import Modules const http = require('http'); const fs = require('fs'); const querystring = require('querystring'); // Create Server http.createServer((req, res) => { // Display Form if (req.method === "GET") { res.writeHead(200, { 'Content-Type': 'text/html' }); res.write(`

Append File Contents

First File Name:

Second File Name:

`); res.end(); } // Handle Form Submission else if (req.method === "POST") { let body = ""; req.on("data", chunk => { body += chunk.toString(); }); req.on("end", () => { let data = querystring.parse(body); let file1 = data.file1; let file2 = data.file2; // Read first file fs.readFile(file1, 'utf8', (err, content) => { if (err) { res.end("Error reading first file"); return; } // Append content to second file fs.appendFile(file2, content, (err) => { if (err) { res.end("Error appending to second file"); return; } res.end("Contents of first file appended to second file successfully!"); }); }); }); } }).listen(3000); console.log("Server running at http://localhost:3000"); ---
