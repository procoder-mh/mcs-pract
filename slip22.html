# Slip 22 ## Q1) Employee Registration Validation // file: employee.js // Import HTTP module const http = require('http'); const querystring = require('querystring'); // Create Server http.createServer((req, res) => { // Display Form if (req.method === "GET") { res.writeHead(200, { 'Content-Type': 'text/html' }); res.write(`

Employee Registration Form

Employee Name:

Email:

Age:

`); res.end(); } // Handle Form Submission else if (req.method === "POST") { let body = ""; req.on("data", chunk => { body += chunk.toString(); }); req.on("end", () => { let data = querystring.parse(body); let ename = data.ename; let email = data.email; let age = data.age; // Validation Patterns let namePattern = /^[A-Za-z ]+$/; let emailPattern = /^[^ ]+@[^ ]+\.[a-z]{2,3}$/; // Name Validation if (!namePattern.test(ename)) { res.end("Invalid Employee Name"); return; } // Email Validation if (!emailPattern.test(email)) { res.end("Invalid Email ID"); return; } // Age Validation if (age < 18 || age > 60) { res.end("Age should be between 18 and 60"); return; } res.end("Employee Registration Successful!"); }); } }).listen(3000); console.log("Server running at http://localhost:3000"); ## Q2) Event Driven Application // file: main.js // Import EventEmitter module const EventEmitter = require('events'); // Create EventEmitter object const eventEmitter = new EventEmitter(); // Callback Function const eventHandler = () => { console.log("Event detected!"); console.log("Callback function executed successfully."); }; // Listen for event named 'message' eventEmitter.on('message', eventHandler); // Main Loop console.log("Application is running..."); console.log("Waiting for events..."); // Trigger Event after 3 seconds setTimeout(() => { console.log("Event Occurred!"); // Emit Event eventEmitter.emit('message'); }, 3000); ---
